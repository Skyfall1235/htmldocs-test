162 void UpdateValue()
163 {
164 var lookDirection = GetLookDirection();
165 var lookAngle = Mathf.Atan2(lookDirection.z, lookDirection.y) * Mathf.Rad2Deg;
166
167 if (m_MinAngle < m_MaxAngle)
168 lookAngle = Mathf.Clamp(lookAngle, m_MinAngle, m_MaxAngle);
169 else
170 lookAngle = Mathf.Clamp(lookAngle, m_MaxAngle, m_MinAngle);
171
172 var maxAngleDistance = Mathf.Abs(m_MaxAngle - lookAngle);
173 var minAngleDistance = Mathf.Abs(m_MinAngle - lookAngle);
174
175 if (m_Value)
176 maxAngleDistance *= (1.0f - k_LeverDeadZone);
177 else
178 minAngleDistance *= (1.0f - k_LeverDeadZone);
179
180 var newValue = (maxAngleDistance < minAngleDistance);
181
182 SetHandleAngle(lookAngle);
183
184 SetValue(newValue);
185 }
