442 #region Calculate Error
443 [MethodImpl(MethodImplOptions.AggressiveInlining)]
444 private static double VertexError(ref SymmetricMatrix q, double x, double y, double z)
445 {
446 return q.m0 * x * x + 2 * q.m1 * x * y + 2 * q.m2 * x * z + 2 * q.m3 * x + q.m4 * y * y
447 + 2 * q.m5 * y * z + 2 * q.m6 * y + q.m7 * z * z + 2 * q.m8 * z + q.m9;
448 }
449
450 [MethodImpl(MethodImplOptions.AggressiveInlining)]
451 private double CurvatureError(ref Vertex vert0, ref Vertex vert1)
452 {
453 double diffVector = (vert0.p - vert1.p).Magnitude;
454
455 var trianglesWithViOrVjOrBoth = triangleHashSet1;
456 trianglesWithViOrVjOrBoth.Clear();
457 GetTrianglesContainingVertex(ref vert0, trianglesWithViOrVjOrBoth);
458 GetTrianglesContainingVertex(ref vert1, trianglesWithViOrVjOrBoth);
459
460 var trianglesWithViAndVjBoth = triangleHashSet2;
461 trianglesWithViAndVjBoth.Clear();
462 GetTrianglesContainingBothVertices(ref vert0, ref vert1, trianglesWithViAndVjBoth);
463
464 double maxDotOuter = 0;
465 foreach (var triangleWithViOrVjOrBoth in trianglesWithViOrVjOrBoth)
466 {
467 double maxDotInner = 0;
468 Vector3d normVecTriangleWithViOrVjOrBoth = triangleWithViOrVjOrBoth.n;
469
470 foreach (var triangleWithViAndVjBoth in trianglesWithViAndVjBoth)
471 {
472 Vector3d normVecTriangleWithViAndVjBoth = triangleWithViAndVjBoth.n;
473 double dot = Vector3d.Dot(ref normVecTriangleWithViOrVjOrBoth, ref normVecTriangleWithViAndVjBoth);
474
475 if (dot > maxDotInner)
476 maxDotInner = dot;
477 }
478
479 if (maxDotInner > maxDotOuter)
480 maxDotOuter = maxDotInner;
481 }
482
483 return diffVector * maxDotOuter;
484 }

486 private double CalculateError(ref Vertex vert0, ref Vertex vert1, out Vector3d result)
487 {
488 // compute interpolated vertex
489 SymmetricMatrix q = (vert0.q + vert1.q);
490 bool borderEdge = (vert0.borderEdge && vert1.borderEdge);
491 double error = 0.0;
492 double det = q.Determinant1();
493 if (det != 0.0 && !borderEdge)
494 {
495 // q_delta is invertible
496 result = new Vector3d(
497 -1.0 / det * q.Determinant2(), // vx = A41/det(q_delta)
498 1.0 / det * q.Determinant3(), // vy = A42/det(q_delta)
499 -1.0 / det * q.Determinant4()); // vz = A43/det(q_delta)
500
501 double curvatureError = 0;
502 if (simplificationOptions.PreserveSurfaceCurvature)
503 {
504 curvatureError = CurvatureError(ref vert0, ref vert1);
505 }
506
507 error = VertexError(ref q, result.x, result.y, result.z) + curvatureError;
508 }
509 else
510 {
511 // det = 0 -> try to find best result
512 Vector3d p1 = vert0.p;
513 Vector3d p2 = vert1.p;
514 Vector3d p3 = (p1 + p2) * 0.5f;
515 double error1 = VertexError(ref q, p1.x, p1.y, p1.z);
516 double error2 = VertexError(ref q, p2.x, p2.y, p2.z);
517 double error3 = VertexError(ref q, p3.x, p3.y, p3.z);
518
519 if (error1 < error2)
520 {
521 if (error1 < error3)
522 {
523 error = error1;
524 result = p1;
525 }
526 else
527 {
528 error = error3;
529 result = p3;
530 }
531 }
532 else if (error2 < error3)
533 {
534 error = error2;
535 result = p2;
536 }
537 else
538 {
539 error = error3;
540 result = p3;
541 }
542 }
543 return error;
544 }

547 #region Calculate Barycentric Coordinates
548 private static void CalculateBarycentricCoords(ref Vector3d point, ref Vector3d a, ref Vector3d b, ref Vector3d c, out Vector3 result)
549 {
550 Vector3d v0 = (b - a), v1 = (c - a), v2 = (point - a);
551 double d00 = Vector3d.Dot(ref v0, ref v0);
552 double d01 = Vector3d.Dot(ref v0, ref v1);
553 double d11 = Vector3d.Dot(ref v1, ref v1);
554 double d20 = Vector3d.Dot(ref v2, ref v0);
555 double d21 = Vector3d.Dot(ref v2, ref v1);
556 double denom = d00 * d11 - d01 * d01;
557
558 // Make sure the denominator is not too small to cause math problems
559 if (Math.Abs(denom) < DenomEpilson)
560 {
561 denom = DenomEpilson;
562 }
563
564 double v = (d11 * d20 - d01 * d21) / denom;
565 double w = (d00 * d21 - d01 * d20) / denom;
566 double u = 1.0 - v - w;
567 result = new Vector3((float)u, (float)v, (float)w);
568 }

581 #region Flipped
585 private bool Flipped(ref Vector3d p, int i0, int i1, ref Vertex v0, bool[] deleted)
586 {
587 int tcount = v0.tcount;
588 var refs = this.refs.Data;
589 var triangles = this.triangles.Data;
590 var vertices = this.vertices.Data;
591 for (int k = 0; k < tcount; k++)
592 {
593 Ref r = refs[v0.tstart + k];
594 if (triangles[r.tid].deleted)
595 continue;
596
597 int s = r.tvertex;
598 int id1 = triangles[r.tid][(s + 1) % 3];
599 int id2 = triangles[r.tid][(s + 2) % 3];
600 if (id1 == i1 || id2 == i1)
601 {
602 deleted[k] = true;
603 continue;
604 }
605
606 Vector3d d1 = vertices[id1].p - p;
607 d1.Normalize();
608 Vector3d d2 = vertices[id2].p - p;
609 d2.Normalize();
610 double dot = Vector3d.Dot(ref d1, ref d2);
611 if (Math.Abs(dot) > 0.999)
612 return true;
613
614 Vector3d n;
615 Vector3d.Cross(ref d1, ref d2, out n);
616 n.Normalize();
617 deleted[k] = false;
618 dot = Vector3d.Dot(ref n, ref triangles[r.tid].n);
619 if (dot < 0.2)
620 return true;
621 }
622
623 return false;
624 }

631 private void UpdateTriangles(int i0, int ia0, ref Vertex v, ResizableArray<bool> deleted, ref int deletedTriangles)
632 {
633 Vector3d p;
634 int tcount = v.tcount;
635 var triangles = this.triangles.Data;
636 var vertices = this.vertices.Data;
637 for (int k = 0; k < tcount; k++)
638 {
639 Ref r = refs[v.tstart + k];
640 int tid = r.tid;
641 Triangle t = triangles[tid];
642 if (t.deleted)
643 continue;
644
645 if (deleted[k])
646 {
647 triangles[tid].deleted = true;
648 ++deletedTriangles;
649 continue;
650 }
651
652 t[r.tvertex] = i0;
653 if (ia0 != -1)
654 {
655 t.SetAttributeIndex(r.tvertex, ia0);
656 }
657
658 t.dirty = true;
659 t.err0 = CalculateError(ref vertices[t.v0], ref vertices[t.v1], out p);
660 t.err1 = CalculateError(ref vertices[t.v1], ref vertices[t.v2], out p);
661 t.err2 = CalculateError(ref vertices[t.v2], ref vertices[t.v0], out p);
662 t.err3 = MathHelper.Min(t.err0, t.err1, t.err2);
663 triangles[tid] = t;
664 refs.Add(r);
665 }

769 #region Remove Vertex Pass
773 private void RemoveVertexPass(int startTrisCount, int targetTrisCount, double threshold, ResizableArray<bool> deleted0, ResizableArray<bool> deleted1, ref int deletedTris)
774 {
775 var triangles = this.triangles.Data;
776 int triangleCount = this.triangles.Length;
777 var vertices = this.vertices.Data;
778
779 Vector3d p;
780 Vector3 barycentricCoord;
781 for (int tid = 0; tid < triangleCount; tid++)
782 {
783 if (triangles[tid].dirty || triangles[tid].deleted || triangles[tid].err3 > threshold)
784 continue;
785
786 triangles[tid].GetErrors(errArr);
787 triangles[tid].GetAttributeIndices(attributeIndexArr);
788 for (int edgeIndex = 0; edgeIndex < TriangleEdgeCount; edgeIndex++)
789 {
790 if (errArr[edgeIndex] > threshold)
791 continue;
792
793 int nextEdgeIndex = ((edgeIndex + 1) % TriangleEdgeCount);
794 int i0 = triangles[tid][edgeIndex];
795 int i1 = triangles[tid][nextEdgeIndex];
796
797 // Border check
798 if (vertices[i0].borderEdge != vertices[i1].borderEdge)
799 continue;
800 // Seam check
801 else if (vertices[i0].uvSeamEdge != vertices[i1].uvSeamEdge)
802 continue;
803 // Foldover check
804 else if (vertices[i0].uvFoldoverEdge != vertices[i1].uvFoldoverEdge)
805 continue;
806 // If borders should be preserved
807 else if (simplificationOptions.PreserveBorderEdges && vertices[i0].borderEdge)
808 continue;
809 // If seams should be preserved
810 else if (simplificationOptions.PreserveUVSeamEdges && vertices[i0].uvSeamEdge)
811 continue;
812 // If foldovers should be preserved
813 else if (simplificationOptions.PreserveUVFoldoverEdges && vertices[i0].uvFoldoverEdge)
814 continue;
815
816 // Compute vertex to collapse to
817 CalculateError(ref vertices[i0], ref vertices[i1], out p);
818 deleted0.Resize(vertices[i0].tcount); // normals temporarily
819 deleted1.Resize(vertices[i1].tcount); // normals temporarily
820
821 // Don't remove if flipped
822 if (Flipped(ref p, i0, i1, ref vertices[i0], deleted0.Data))
823 continue;
824 if (Flipped(ref p, i1, i0, ref vertices[i1], deleted1.Data))
825 continue;
826
827 // Calculate the barycentric coordinates within the triangle
828 int nextNextEdgeIndex = ((edgeIndex + 2) % 3);
829 int i2 = triangles[tid][nextNextEdgeIndex];
830 CalculateBarycentricCoords(ref p, ref vertices[i0].p, ref vertices[i1].p, ref vertices[i2].p, out barycentricCoord);
831
832 // Not flipped, so remove edge
833 vertices[i0].p = p;
834 vertices[i0].q += vertices[i1].q;
835
836 // Interpolate the vertex attributes
837 int ia0 = attributeIndexArr[edgeIndex];
838 int ia1 = attributeIndexArr[nextEdgeIndex];
839 int ia2 = attributeIndexArr[nextNextEdgeIndex];
840 InterpolateVertexAttributes(ia0, ia0, ia1, ia2, ref barycentricCoord);
841
842 if (vertices[i0].uvSeamEdge)
843 {
844 ia0 = -1;
845 }
846
847 int tstart = refs.Length;
848 UpdateTriangles(i0, ia0, ref vertices[i0], deleted0, ref deletedTris);
849 UpdateTriangles(i0, ia0, ref vertices[i1], deleted1, ref deletedTris);
850
851 int tcount = refs.Length - tstart;
852 if (tcount <= vertices[i0].tcount)
853 {
854 // save ram
855 if (tcount > 0)
856 {
857 var refsArr = refs.Data;
858 Array.Copy(refsArr, tstart, refsArr, vertices[i0].tstart, tcount);
859 }
860 }
861 else
862 {
863 // append
864 vertices[i0].tstart = tstart;
865 }
866
867 vertices[i0].tcount = tcount;
868 break;
869 }
870
871 // Check if we are already done
872 if ((startTrisCount - deletedTris) <= targetTrisCount)
873 break;
874 }

883 private void UpdateMesh(int iteration)
884 {
885 var triangles = this.triangles.Data;
886 var vertices = this.vertices.Data;
887
888 int triangleCount = this.triangles.Length;
889 int vertexCount = this.vertices.Length;
890 if (iteration > 0) // compact triangles
891 {
892 int dst = 0;
893 for (int i = 0; i < triangleCount; i++)
894 {
895 if (!triangles[i].deleted)
896 {
897 if (dst != i)
898 {
899 triangles[dst] = triangles[i];
900 triangles[dst].index = dst;
901 }
902 dst++;
903 }
904 }
905 this.triangles.Resize(dst);
906 triangles = this.triangles.Data;
907 triangleCount = dst;
908 }
909
910 UpdateReferences();
911
912 // Identify boundary : vertices[].border=0,1
913 if (iteration == 0)
914 {
915 var refs = this.refs.Data;
916
917 var vcount = new List<int>(8);
918 var vids = new List<int>(8);
919 int vsize = 0;
920 for (int i = 0; i < vertexCount; i++)
921 {
922 vertices[i].borderEdge = false;
923 vertices[i].uvSeamEdge = false;
924 vertices[i].uvFoldoverEdge = false;
925 }
926
927 int ofs;
928 int id;
929 int borderVertexCount = 0;
930 double borderMinX = double.MaxValue;
931 double borderMaxX = double.MinValue;
932 var vertexLinkDistanceSqr = simplificationOptions.VertexLinkDistance * simplificationOptions.VertexLinkDistance;
933 for (int i = 0; i < vertexCount; i++)
934 {
935 int tstart = vertices[i].tstart;
936 int tcount = vertices[i].tcount;
937 vcount.Clear();
938 vids.Clear();
939 vsize = 0;
940
941 for (int j = 0; j < tcount; j++)
942 {
943 int tid = refs[tstart + j].tid;
944 for (int k = 0; k < TriangleVertexCount; k++)
945 {
946 ofs = 0;
947 id = triangles[tid][k];
948 while (ofs < vsize)
949 {
950 if (vids[ofs] == id)
951 break;
952
953 ++ofs;
954 }
955
956 if (ofs == vsize)
957 {
958 vcount.Add(1);
959 vids.Add(id);
960 ++vsize;
961 }
962 else
963 {
964 ++vcount[ofs];
965 }
966 }
967 }
968
969 for (int j = 0; j < vsize; j++)
970 {
971 if (vcount[j] == 1)
972 {
973 id = vids[j];
974 vertices[id].borderEdge = true;
975 ++borderVertexCount;
976
977 if (simplificationOptions.EnableSmartLink)
978 {
979 if (vertices[id].p.x < borderMinX)
980 {
981 borderMinX = vertices[id].p.x;
982 }
983 if (vertices[id].p.x > borderMaxX)
984 {
985 borderMaxX = vertices[id].p.x;
986 }
987 }
988 }
989 }
990 }
991
992 if (simplificationOptions.EnableSmartLink)
993 {
994 // First find all border vertices
995 var borderVertices = new BorderVertex[borderVertexCount];
996 int borderIndexCount = 0;
997 double borderAreaWidth = borderMaxX - borderMinX;
998 for (int i = 0; i < vertexCount; i++)
999 {
1000 if (vertices[i].borderEdge)
1001 {
1002 int vertexHash = (int)(((((vertices[i].p.x - borderMinX) / borderAreaWidth) * 2.0) - 1.0) * int.MaxValue);
1003 borderVertices[borderIndexCount] = new BorderVertex(i, vertexHash);
1004 ++borderIndexCount;
1005 }
1006 }
1007
1008 // Sort the border vertices by hash
1009 Array.Sort(borderVertices, 0, borderIndexCount, BorderVertexComparer.instance);
1010
1011 // Calculate the maximum hash distance based on the maximum vertex link distance
1012 double vertexLinkDistance = Math.Sqrt(vertexLinkDistanceSqr);
1013 int hashMaxDistance = Math.Max((int)((vertexLinkDistance / borderAreaWidth) * int.MaxValue), 1);
1014
1015 // Then find identical border vertices and bind them together as one
1016 for (int i = 0; i < borderIndexCount; i++)
1017 {
1018 int myIndex = borderVertices[i].index;
1019 if (myIndex == -1)
1020 continue;
1021
1022 var myPoint = vertices[myIndex].p;
1023 for (int j = i + 1; j < borderIndexCount; j++)
1024 {
1025 int otherIndex = borderVertices[j].index;
1026 if (otherIndex == -1)
1027 continue;
1028 else if ((borderVertices[j].hash - borderVertices[i].hash) > hashMaxDistance) // There is no point to continue beyond this point
1029 break;
1030
1031 var otherPoint = vertices[otherIndex].p;
1032 var sqrX = ((myPoint.x - otherPoint.x) * (myPoint.x - otherPoint.x));
1033 var sqrY = ((myPoint.y - otherPoint.y) * (myPoint.y - otherPoint.y));
1034 var sqrZ = ((myPoint.z - otherPoint.z) * (myPoint.z - otherPoint.z));
1035 var sqrMagnitude = sqrX + sqrY + sqrZ;
1036
1037 if (sqrMagnitude <= vertexLinkDistanceSqr)
1038 {
1039 borderVertices[j].index = -1; // NOTE: This makes sure that the "other" vertex is not processed again
1040 vertices[myIndex].borderEdge = false;
1041 vertices[otherIndex].borderEdge = false;
1042
1043 if (AreUVsTheSame(0, myIndex, otherIndex))
1044 {
1045 vertices[myIndex].uvFoldoverEdge = true;
1046 vertices[otherIndex].uvFoldoverEdge = true;
1047 }
1048 else
1049 {
1050 vertices[myIndex].uvSeamEdge = true;
1051 vertices[otherIndex].uvSeamEdge = true;
1052 }
1053
1054 int otherTriangleCount = vertices[otherIndex].tcount;
1055 int otherTriangleStart = vertices[otherIndex].tstart;
1056 for (int k = 0; k < otherTriangleCount; k++)
1057 {
1058 var r = refs[otherTriangleStart + k];
1059 triangles[r.tid][r.tvertex] = myIndex;
1060 }
1061 }
1062 }
1063 }
1064
1065 // Update the references again
1066 UpdateReferences();
1067 }
1068
1069 // Init Quadrics by Plane & Edge Errors
1070 //
1071 // required at the beginning ( iteration == 0 )
1072 // recomputing during the simplification is not required,
1073 // but mostly improves the result for closed meshes
1074 for (int i = 0; i < vertexCount; i++)
1075 {
1076 vertices[i].q = new SymmetricMatrix();
1077 }
1078
1079 int v0, v1, v2;
1080 Vector3d n, p0, p1, p2, p10, p20, dummy;
1081 SymmetricMatrix sm;
1082 for (int i = 0; i < triangleCount; i++)
1083 {
1084 v0 = triangles[i].v0;
1085 v1 = triangles[i].v1;
1086 v2 = triangles[i].v2;
1087
1088 p0 = vertices[v0].p;
1089 p1 = vertices[v1].p;
1090 p2 = vertices[v2].p;
1091 p10 = p1 - p0;
1092 p20 = p2 - p0;
1093 Vector3d.Cross(ref p10, ref p20, out n);
1094 n.Normalize();
1095 triangles[i].n = n;
1096
1097 sm = new SymmetricMatrix(n.x, n.y, n.z, -Vector3d.Dot(ref n, ref p0));
1098 vertices[v0].q += sm;
1099 vertices[v1].q += sm;
1100 vertices[v2].q += sm;
1101 }
1102
1103 for (int i = 0; i < triangleCount; i++)
1104 {
1105 // Calc Edge Error
1106 var triangle = triangles[i];
1107 triangles[i].err0 = CalculateError(ref vertices[triangle.v0], ref vertices[triangle.v1], out dummy);
1108 triangles[i].err1 = CalculateError(ref vertices[triangle.v1], ref vertices[triangle.v2], out dummy);
1109 triangles[i].err2 = CalculateError(ref vertices[triangle.v2], ref vertices[triangle.v0], out dummy);
1110 triangles[i].err3 = MathHelper.Min(triangles[i].err0, triangles[i].err1, triangles[i].err2);
1111 }
1112 }
1113 }
