85 private async Task FetchHistory(bool scrollToBottom = false)
86#pragma warning restore CS1998 // Async method lacks 'await' operators and will run synchronously
87 {
88#if !UNITY_WEBGL
89 try
90 {
91 var chatHistoryOptions = new ChatHistoryQueryOptions()
92 {
93 TimeEnd = oldestMessage
94 };
95 var historyMessages =
96 await VivoxService.Instance.GetChannelTextMessageHistoryAsync(VivoxVoiceManager.LobbyChannelName, 10,
97 chatHistoryOptions);
98 var reversedMessages = historyMessages.Reverse();
99 foreach (var historyMessage in reversedMessages)
100 {
101 AddMessageToChat(historyMessage, true, scrollToBottom);
102 }
103
104 // Update the oldest message ReceivedTime if it exists to help the next fetch get the next batch of history
105 oldestMessage = historyMessages.FirstOrDefault()?.ReceivedTime;
106 }
107 catch (TaskCanceledException e)
108 {
109 Debug.Log($"Chat history request was canceled, likely because of a logout or the data is no longer needed: {e.Message}");
110 }
111 catch (Exception e)
112 {
113 Debug.LogError($"Tried to fetch chat history and failed with error: {e.Message}");
114 }

238 void AddMessageToChat(VivoxMessage message, bool isHistory = false, bool scrollToBottom = false)
239 {
240 var newMessageObj = Instantiate(MessageObject, ChatContentObj.transform);
241 var newMessageTextObject = newMessageObj.GetComponent<MessageObjectUI>();
242 if (isHistory)
243 {
244 m_MessageObjPool.Insert(0, new KeyValuePair<string, MessageObjectUI>(message.MessageId, newMessageTextObject));
245 newMessageObj.transform.SetSiblingIndex(0);
246 }
247 else
248 {
249 m_MessageObjPool.Add(new KeyValuePair<string, MessageObjectUI>(message.MessageId, newMessageTextObject));
250 }
251
252 newMessageTextObject.SetTextMessage(message);
253 if (scrollToBottom)
254 {
255 StartCoroutine(SendScrollRectToBottom());
256 }
257
258 if (!message.FromSelf)
259 {
260 if (ToggleTTS.isOn)
261 {
262 VivoxService.Instance.TextToSpeechSendMessage($"{message.SenderDisplayName} said, {message.MessageText}", TextToSpeechMessageType.LocalPlayback);
263 }
