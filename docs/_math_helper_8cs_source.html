69 #region Min
77 [MethodImpl(MethodImplOptions.AggressiveInlining)]
78 public static double Min(double val1, double val2, double val3)
79 {
80 return (val1 < val2 ? (val1 < val3 ? val1 : val3) : (val2 < val3 ? val2 : val3));
81 }
82 #endregion
83
84 #region Clamping
92 [MethodImpl(MethodImplOptions.AggressiveInlining)]
93 public static double Clamp(double value, double min, double max)
94 {
95 return (value >= min ? (value <= max ? value : max) : min);
96 }
97 #endregion
98
99 #region Triangle Area
107 [MethodImpl(MethodImplOptions.AggressiveInlining)]
108 public static double TriangleArea(ref Vector3d p0, ref Vector3d p1, ref Vector3d p2)
109 {
110 var dx = p1 - p0;
111 var dy = p2 - p0;
112 return dx.Magnitude * (Math.Sin(Vector3d.Angle(ref dx, ref dy) * Deg2Radd) * dy.Magnitude) * 0.5f;
113 }
114 #endregion
115 }
