170 public virtual void ShootFunc()
171 {
172 //If the weapon has ammo in the magazine and hasn't been fired yet, then we can shoot the gun
173 if (ammo > 0 && !weaponFired)
174 {
175 //begin the haptic feedback/recoil and start the countdown for the firerate
176 Cooldown();
177
178 RaycastHit hitInfo;
179
180 //Attaches the bullet tracer to the "bullet"
181 StartTracer(muzzle.transform.forward);
182 Debug.DrawRay(muzzle.transform.position, muzzle.transform.forward * weaponInfo.range, Color.red, 10.0f);
183
184 //Send a rayCast straight out from the tip of the barrel. (stores the information of any object we hit in hitInfo)
185 if (Physics.Raycast(muzzle.transform.position, muzzle.transform.forward, out hitInfo, weaponInfo.range))
186 {
187 Debug.Log("Hit " + hitInfo.transform.name);
188
189
190 StartImpactParticle(hitInfo);
191
192
193 //Check to see if we hit an enemy (The enemy damage regisration component is called IDamageable)
194
195 if (hitInfo.transform.TryGetComponent<HealthManager>(out HealthManager enemy))
196 {
197 weaponInfo.damageData.DamageAmount = (ushort)weaponInfo.damageSO.CalculateDamage(damage, hitInfo, damageFallOffStart);
198 enemy.Damage(weaponInfo.damageData);
199 }
200
201
202
203 if (hitInfo.transform.TryGetComponent(out HitDetection target))
204 {
205 if (target != null)
206 target.OnHit(weaponInfo.damageSO.CalculateDamage(damage,hitInfo,damageFallOffStart), 1); //If an enemy was hit then calculate the amount of damage that should be done
207 }
208 }
209 }
