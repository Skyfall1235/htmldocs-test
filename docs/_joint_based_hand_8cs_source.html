157 List<JointToTransformReference> GetFingerJoints(string fingerName, int startDepth, int jointDepth, XRHandJointID fingerStartJointId)
158 {
159 try
160 {
161 List<JointToTransformReference> fingerJoints = new();
162 JointToTransformReference currentJoint = new();
163
164 foreach (Transform child in m_HandRoot)
165 {
166 if (child.name.Contains(fingerName))
167 {
168 Transform currentChild = child;
169
170 //Navigate to the starting joint based on the startDepth
171 for(int i = 0; i < startDepth; i++)
172 {
173 currentChild = currentChild.GetChild(0);
174 }
175
176 // Get all joints in the finger and add them to the list based on the jointDepth
177 for (int i = 0; i < jointDepth; i++)
178 {
179 currentJoint.jointTransform = currentChild;
180 int currentHandJointId = (int)fingerStartJointId + i + startDepth;
181 currentJoint.xrHandJointID = (XRHandJointID)currentHandJointId;
182 fingerJoints.Add(currentJoint);
183 currentChild = currentChild.GetChild(0);
184 }
185 }
