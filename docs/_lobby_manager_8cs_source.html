119 public async Task<Lobby> JoinLobby(Lobby lobby = null, string roomCode = null)

136 if (e is LobbyServiceException)
137 {
138 string message = e.Message.ToLower();
139
140 if (message.Contains("Rate limit".ToLower()))
141 failureMessage = "Rate limit exceeded. Please try again later.";
142 else if (message.Contains("Lobby not found".ToLower()))
143 failureMessage = "Lobby not found. Please try a new Lobby.";
144 else
145 failureMessage = e.Message;
146 }
147 Utils.Log($"{k_DebugPrepend}{failureMessage}\n\n{e}", 1);
148 OnLobbyFailed?.Invoke($"{failureMessage}");
149 return null;

157 public async Task<Lobby> CreateLobby(string roomName = null, bool isPrivate = false, int playerCount = XRINetworkGameManager.maxPlayers)
158 {
159 try
160 {
161 m_Status.Value = "Creating Relay";
162 // Creates a new Allocation based on the defined max players above
163 var alloc = await RelayService.Instance.CreateAllocationAsync(XRINetworkGameManager.maxPlayers);
164
165 m_Status.Value = "Creating Join Code";
166 // Get a join code based on the Allocation
167 var joinCode = await RelayService.Instance.GetJoinCodeAsync(alloc.AllocationId);
168
169 // Creates Lobby Options Dictionary for other clients to find and join
170 var options = new CreateLobbyOptions
171 {
172 // Set the Data to be used for lobby filtering
173 Data = new Dictionary<string, DataObject>
174 {
175 {
176 // Set Join Code Key
177 k_JoinCodeKeyIdentifier, new DataObject(DataObject.VisibilityOptions.Public, joinCode)
178 },
179 {
180 // Set Region Key
181 k_RegionKeyIdentifier, new DataObject(DataObject.VisibilityOptions.Public, alloc.Region)
182 },
183 {
184 // Set Build ID Key
185 k_BuildIdKeyIdentifier, new DataObject(DataObject.VisibilityOptions.Public, Application.version, DataObject.IndexOptions.S1)
186 },
187 {
188 // Set Scene Key
189 k_SceneKeyIdentifier, new DataObject(DataObject.VisibilityOptions.Public, SceneManager.GetActiveScene().name, DataObject.IndexOptions.S2)
190 },
191 {
192 // Set Editor Key
193 k_EditorKeyIdentifier, new DataObject(DataObject.VisibilityOptions.Public, hideEditorFromLobby.ToString(), DataObject.IndexOptions.S3)
194 },
195 },
196 IsPrivate = isPrivate,
197 };
198
199
200 m_Status.Value = "Creating Lobby";
201 // Creates the Lobby with the specified max players and lobby options. Currently just naming "General Lobby"
202 string lobbyName = string.IsNullOrEmpty(roomName) ? $"{XRINetworkGameManager.LocalPlayerName.Value}'s Room" : $"{roomName}";
203
204 // RATE LIMIT: 2 request per 6 seconds
205 var lobby = await LobbyService.Instance.CreateLobbyAsync(lobbyName, playerCount, options);
206 Utils.Log($"{k_DebugPrepend}Created Lobby with Join Code: {joinCode}, Region: {alloc.Region}, Build ID: {Application.version}, Scene: {SceneManager.GetActiveScene().name}, Editor: {hideEditorFromLobby}");
207
208 // Stop the heartbeat routine if one exists, and starts a new one. This keeps the lobby active for visibility
209 if (m_HeartBeatRoutine != null) StopCoroutine(m_HeartBeatRoutine);
210 m_HeartBeatRoutine = StartCoroutine(LobbyHeartbeatCoroutine(lobby.Id));
211
212 //Populate the transport data with the relay info for the host (IP, port, etc...)
213 m_Transport.SetHostRelayData(alloc.RelayServer.IpV4, (ushort)alloc.RelayServer.Port, alloc.AllocationIdBytes, alloc.Key, alloc.ConnectionData);
214 ConnectedToLobby(lobby);
215
216 return lobby;
217 }
218 catch (Exception e)
219 {
220 string failureMessage = "Failed to Create Lobby. Please try again.";
221 Utils.Log($"{k_DebugPrepend}{failureMessage}\n\n{e}", 1);
222 // Debug.LogWarning($"[XRMPT] {failureMessage}\n\n{e}");
223 OnLobbyFailed?.Invoke(failureMessage);
224 return null;
225 }
226 }

238 );

311 IEnumerator LobbyHeartbeatCoroutine(string lobbyId, float waitTimeSeconds = 15.0f)
312 {
313 // Setup a new wait based on wait time in seconds
314 var delay = new WaitForSecondsRealtime(waitTimeSeconds);
315
316 while (true)
317 {
318 // Continuously ping the lobby to keep it alive
319 LobbyService.Instance.SendHeartbeatPingAsync(lobbyId);

321 yield return delay;

455 public static bool CheckForLobbyFilter(Lobby lobby)
456 {
457 // If the lobby is not in the same scene, skip it
458 if (lobby.Data.TryGetValue(k_SceneKeyIdentifier, out DataObject sceneData))
459 {
460 if (sceneData.Value != SceneManager.GetActiveScene().name)
461 {
462 return true;
463 }
464 }
465
466 if (lobby.Data.TryGetValue(k_EditorKeyIdentifier, out DataObject editorData))
467 {
468 // If the lobby is an editor lobby is set to filter return true
469 if (editorData.Value == "True" & !s_HideEditorInLobbies)
470 {
471 return true;
472 }
473 }
474
475 return false;
