589 public override void OnGainedOwnership()
590 {
591 base.OnGainedOwnership();
592
593 // Check for gaining ownership of an object when a player disconnects
594 if (IsOwner && IsServer && isInteracting & !baseInteractable.isSelected)
595 {
596 m_IsInteracting.Value = false;
597 }
598
599 // Workaround for NGO calling this always on Server, even if Server is not owner.
600 // So we check IsOwner and Interactable selected state, and loop through to check for sockets.
601 if (IsOwner)
602 {
603 if (m_HostInteractionCheckEnumerator != null)
604 StopCoroutine(m_HostInteractionCheckEnumerator);
605
606 m_HostInteractionCheckEnumerator = CheckForOwnerInteraction();
607 StartCoroutine(m_HostInteractionCheckEnumerator);
608
609 if (m_RelinquishToHostEnumerator != null) StopCoroutine(m_RelinquishToHostEnumerator);
610
611 if (baseInteractable.isSelected & !isInteracting)
612 {
613 if (!IsSelectedBySocket())
614 {
615 m_IsInteracting.Value = true;
616 }
617 }
618 }
619 }

624 public override void OnLostOwnership()
625 {
626 base.OnLostOwnership();
627
628 // Have to check ownership since this is always called on the Server
629 if (!IsOwner)
630 {
631 if (m_HostInteractionCheckEnumerator != null)
632 StopCoroutine(m_HostInteractionCheckEnumerator);
633
634 if (m_RelinquishToHostEnumerator != null)
635 StopCoroutine(m_RelinquishToHostEnumerator);
636
637 if (baseInteractable.isSelected)
638 m_InteractionManager.CancelInteractableSelection((IXRSelectInteractable)baseInteractable);
639
640 }

646 IEnumerator CheckForOwnerInteraction()
647 {
648 while (IsOwner)
649 {
650 if (isInteracting)
651 {
652 if (!baseInteractable.isSelected || IsSelectedBySocket())
653 {
654 Utils.Log($"Interacting is true and either selected is false or selected is a socket on object {gameObject.name}. Is this intentional?");
655 m_IsInteracting.Value = false;
656 }
657 }
658 yield return new WaitForSeconds(interactionCheckTime);
659 }
660 }

666 {
667 if (baseInteractable.isSelected)
668 {

715 }
