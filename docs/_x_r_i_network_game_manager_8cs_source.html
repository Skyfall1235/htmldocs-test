464 protected virtual async Task<bool> AbleToConnect()
465 {
466 // If in the process of trying to connect, send failure message and return false.
467 if (m_ConnectionState.Value == ConnectionState.Connecting)
468 {
469 string failureMessage = "Connection attempt still in progress.";
470 Utils.Log($"{k_DebugPrepend}{failureMessage}", 1);
471 ConnectionFailed(failureMessage);
472 return false;
473 }
474
475 // If already connected to a lobby, disconnect in attempt to "Hot Join".
476 if (Connected.Value || m_ConnectionState.Value == ConnectionState.Connected)
477 {
478 Utils.Log($"{k_DebugPrepend}Already Connected to a Lobby. Disconnecting.", 0);
479 await DisconnectAsync();
480
481 // Small wait while everything finishes disconnecting.
482 // This isn't technically needed, but makes the flow feel better.
483 await Task.Delay(100);
484 }
485
486 m_ConnectionState.Value = ConnectionState.Connecting;
487 return true;
488 }

508 protected virtual bool ConnectedToLobby()
509 {
510 bool connected;
511 if (autoConnectOnLobbyJoin)
512 {
513 ConnectedRoomRegion = m_LobbyManager.connectedLobby.Data[LobbyManager.k_RegionKeyIdentifier].Value;
514 ConnectedRoomCode = m_LobbyManager.connectedLobby.LobbyCode;
515 ConnectedRoomName.Value = m_LobbyManager.connectedLobby.Name;
516
517 if (m_LobbyManager.connectedLobby.HostId == AuthenicationId)

521 else

524 }

535 m_ConnectionState.Value = ConnectionState.Connected;

552 protected virtual async void SubscribeToLobbyEvents()
553 {
554 var callbacks = new LobbyEventCallbacks();
555 callbacks.LobbyChanged += OnLobbyChanged;
556 callbacks.LobbyEventConnectionStateChanged += OnLobbyEventConnectionStateChanged;
557 try
558 {
559 await LobbyService.Instance.SubscribeToLobbyEventsAsync(m_LobbyManager.connectedLobby.Id, callbacks);
560 }
561 catch (LobbyServiceException ex)
562 {
563 switch (ex.Reason)
564 {
565 case LobbyExceptionReason.AlreadySubscribedToLobby: Utils.Log($"{k_DebugPrepend}Already subscribed to lobby[{m_LobbyManager.connectedLobby.Id}]. We did not need to try and subscribe again. Exception Message: {ex.Message}", 1); break;
566 case LobbyExceptionReason.SubscriptionToLobbyLostWhileBusy: Utils.Log($"{k_DebugPrepend}Subscription to lobby events was lost while it was busy trying to subscribe. Exception Message: {ex.Message}", 2); throw;
567 case LobbyExceptionReason.LobbyEventServiceConnectionError: Utils.Log($"{k_DebugPrepend}Failed to connect to lobby events. Exception Message: {ex.Message}", 2); throw;
568 default: throw;
569 }
570 }
571 }

577 private void OnLobbyEventConnectionStateChanged(LobbyEventConnectionState state)
578 {
579 switch (state)
580 {
581 case LobbyEventConnectionState.Unsubscribed: Utils.Log($"{k_DebugPrepend}Lobby event now Unsubscribed"); break;
582 case LobbyEventConnectionState.Subscribing: Utils.Log($"{k_DebugPrepend}Attempting to subscribe to lobby events"); break;
583 case LobbyEventConnectionState.Subscribed: Utils.Log($"{k_DebugPrepend}Subscribing to lobby events now"); break;
584 case LobbyEventConnectionState.Unsynced:
585 m_LobbyManager.ReconnectToLobby();
586 Utils.Log($"{k_DebugPrepend}Lobby Events now unsynced.\n\n{state}", 1);
587 break;
588 case LobbyEventConnectionState.Error: Utils.Log($"{k_DebugPrepend}Lobby event error.\n\n{state}", 2); break;
589 }
