251 private void SubTriggerEntered(Collider other)
252 {
253 Rigidbody body = other.GetComponentInParent<Rigidbody>();
254 if (body != null)
255 {
256 if (m_RigidbodyList.Contains(body)) return;
257 // If the object is an interactable and it throws on detach, add it to the list.
258 if (body.TryGetComponent(out UnityEngine.XR.Interaction.Toolkit.Interactables.XRGrabInteractable interactable))
259 {
260 if (interactable.throwOnDetach)
261 {
262 m_RigidbodyList.Add(body);
263 }
264 }
265 else
266 {
267 m_RigidbodyList.Add(body);
268 }
269 }
270 else
271 {
272 // If the object is a character controller, add it to the list.
273 if (other.TryGetComponent(out CharacterController controller))
274 {
275 m_CharacterController = controller;
276 m_CharacterController.GetComponentInChildren<DynamicMoveProvider>().useGravity = false;
277 }
278 }
