321 public void RequestOwnership()
322 {

353 IEnumerator CheckOwnershipRoutine()
354 {
355 // Get the current RTT to the server and wait for twice that time before checking if ownership was granted.
356 float waitTime = NetworkManager.Singleton.NetworkConfig.NetworkTransport.GetCurrentRtt(NetworkManager.ServerClientId) * 2;
357
358 if (waitTime < .025f || waitTime > 5.0f)
359 {
360 waitTime = 1.0f;
361 }
362
363 yield return new WaitForSeconds(waitTime);
364 if (!IsOwner)
365 {
366 Utils.Log($"Ownership Request Timed Out on Object {gameObject.name}");
367 }
368 m_RequestingOwnership = false;
