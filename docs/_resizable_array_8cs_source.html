30namespace UnityMeshSimplifier

36 internal sealed class ResizableArray<T>
37 {
38 #region Fields
39 private T[] items = null;
40 private int length = 0;
41
42 private static T[] emptyArr = new T[0];
43 #endregion
44
45 #region Properties
49 public int Length
50 {
51 [MethodImpl(MethodImplOptions.AggressiveInlining)]
52 get { return length; }
53 }
54
58 public T[] Data
59 {
60 [MethodImpl(MethodImplOptions.AggressiveInlining)]
61 get { return items; }
62 }
63
69 public T this[int index]
70 {
71 [MethodImpl(MethodImplOptions.AggressiveInlining)]
72 get { return items[index]; }
73 [MethodImpl(MethodImplOptions.AggressiveInlining)]
74 set { items[index] = value; }
75 }
76 #endregion

78 #region Constructor
83 public ResizableArray(int capacity)
84 : this(capacity, 0)
85 {
86
87 }
88
94 public ResizableArray(int capacity, int length)
95 {
96 if (capacity < 0)
97 throw new ArgumentOutOfRangeException(nameof(capacity));
98 else if (length < 0 || length > capacity)
99 throw new ArgumentOutOfRangeException(nameof(length));
100
101 if (capacity > 0)
102 items = new T[capacity];
103 else
104 items = emptyArr;
105
106 this.length = length;
107 }
108
113 public ResizableArray(T[] initialArray)
114 {
115 if (initialArray == null)
116 throw new ArgumentNullException(nameof(initialArray));
117
118 if (initialArray.Length > 0)
119 {
120 items = new T[initialArray.Length];
121 length = initialArray.Length;
122 Array.Copy(initialArray, 0, items, 0, initialArray.Length);
123 }
124 else
125 {
126 items = emptyArr;
127 length = 0;
128 }
129 }
130 #endregion
131
132 #region Private Methods
133 private void IncreaseCapacity(int capacity)
134 {
135 T[] newItems = new T[capacity];
136 Array.Copy(items, 0, newItems, 0, System.Math.Min(length, capacity));
137 items = newItems;
138 }
139 #endregion
140
141 #region Public Methods
145 public void Clear()
146 {
147 Array.Clear(items, 0, length);
148 length = 0;
149 }
150
157 public void Resize(int length, bool trimExess = false, bool clearMemory = false)
158 {
159 if (length < 0)
160 throw new ArgumentOutOfRangeException(nameof(length));
161
162 if (length > items.Length)
163 {
164 IncreaseCapacity(length);
165 }
166 else if (length < this.length && clearMemory)
167 {
168 Array.Clear(items, length, this.length - length);
169 }
170
171 this.length = length;
172
173 if (trimExess)
174 {
175 TrimExcess();
176 }
177 }
178
182 public void TrimExcess()
183 {
184 if (items.Length == length) // Nothing to do
185 return;
186
187 var newItems = new T[length];
188 Array.Copy(items, 0, newItems, 0, length);
189 items = newItems;
190 }
191
196 public void Add(T item)
197 {
198 if (length >= items.Length)
199 {
200 IncreaseCapacity(items.Length << 1);
201 }
202
203 items[length++] = item;
204 }
205
210 public T[] ToArray()
211 {
212 var newItems = new T[length];
213 Array.Copy(items, 0, newItems, 0, length);
214 return newItems;
215 }
216 #endregion
217 }
