212 public void Optimize()
213 {
214 m_IsCurrentlyOptimized = true;
215 foreach (var layoutGroup in m_LayoutGroups)

217 layoutGroup.enabled = false;
218 }
219
220 foreach (var keyData in m_KeyData)
221 {
222 var key = keyData.key;
223 if (key == null)
224 continue;
225 key.transform.SetParent(m_ButtonParentTransform);
226
227 if (key.targetGraphic != null)
228 key.targetGraphic.transform.SetParent(m_ImageParentTransform);
229
230 if (key.textComponent != null)
231 key.textComponent.transform.SetParent(m_TextParentTransform);
232
233 if (key.iconComponent != null)
234 key.iconComponent.transform.SetParent(m_IconParentTransform);
235
236 if (key.highlightComponent != null)
237 key.highlightComponent.transform.SetParent(m_HighlightParentTransform);
238
239 if (keyData.batchFollow != null)

247 public void Unoptimize()
