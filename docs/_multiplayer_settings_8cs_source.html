159 public async void SearchForLobbies()
160 {
161 try
162 {
163 QueryLobbiesOptions options = new QueryLobbiesOptions();
164 options.Count = 25;
165
166
167 // Filter for open lobbies only
168 options.Filters = new List<QueryFilter>()
169 {
170 new QueryFilter(
171 field: QueryFilter.FieldOptions.AvailableSlots,
172 op: QueryFilter.OpOptions.GT,
173 value: "0")
174 };
175
176 // Order by newest lobbies first
177 options.Order = new List<QueryOrder>()
178 {
179 new QueryOrder(
180 asc: false,
181 field: QueryOrder.FieldOptions.Created)
182 };
183
184 QueryResponse lobbies = await LobbyService.Instance.QueryLobbiesAsync(options);
185
186 RefreshCurrentLobbies(lobbies);
187 }
188 catch (LobbyServiceException e)
189 {
190 Debug.Log(e);
191 }
192 }
